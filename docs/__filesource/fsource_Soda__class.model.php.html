  38.  * For instance, you can invoke the 'save' method on a Soda model object.

  74.     /**
  75.      * Loads the first object of the table specified in model::$table_name.
  76.      * For example, if your model looks like this:
  77.      * <code>
  78.      * class user extends model {
  79.      *   static $table_name = 'user';
  80.      * }
  81.      * </code>
  82.      *
  83.      * then calling $first_user = user::load() will simply return the first user in the database as a Soda model object.
  84.      *
  85.      *
  86.      * This method uses an optional array of class names to load associated objects.
  87.      *
  88.      * Example: $author = author::load(false, $include = array('book'));
  89.      *
  90.      *
  91.      * You can also specify nested relations:
  92.      *
  93.      * $publisher = publisher::load(false, $include = array('author' => 'book'));
  94.      *
  95.      * This will load all authors and their books as an object hierarchy:
  96.      *
  97.      * <code>
  98.      * $first_author = array_shift($publisher->authors);
  99.      * $books_for_first_author = $first_author->books;
  100.      * </code>
  101.      *
  102.      * Please note that there's also a 'plural' version of this function: model::load_all.
  103.      *
  104.      * @param string $where_clause      SQL where clause: use without the WHERE keyword (optional)
  105.      * @param array  $include           class names of associated models to include (optional)
  106.      * @param array  $params            Array of sql parameters (optional)
  107.      * @return object                   Returns object or null if no object was found or false upon error
  108.      */
  109.     public static function load($where_clause = false, $include = false, $params = null) {
  110.         if (!$objects = static::load_all($where_clause, $include, $params, $limitfrom = -1, $limitnum = 1)) return false;
  111.         return array_shift($objects);
  112.     } // function load

  115.     /**
  116.      * Loads specified associations and attaches them to parent objects.
  117.      *
  118.      * Example: seat::load_associations($seats, $bookings)
  119.      *
  120.      * This will endow each 'seat' object with the property bookings.
  121.      * If a particular seat has no bookings, the property will be an empty array. In turn, each booking object
  122.      * will be endowed with a property 'seat', containing a reference to the parent object.
  123.      *
  124.      * The associations parameter can also be a nested array:
  125.      *
  126.      * array('author' => array('book' => 'chapter'))
  127.      *
  128.      * The association is recognized if your associated child objects contain a foreign key
  129.      * which is named [model_name]_id. The model class for the child objects may also contain
  130.      * a plural version of its name:
  131.      *
  132.      * <code>
  133.      * class mouse extends model {
  134.      *     static $plural = 'mice';
  135.      *     // ...
  136.      * }
  137.      * </code>
  138.      *
  139.      * If you don't specify your own plural version, Soda will simply use an 's' postfix to pluralize.
  140.      *
  141.      * Finally, if the $associations parameter contains a key which points to an object, the properties
  142.      * of this object will be used to impose constraints on the retrieved associated objects. Technically,
  143.      * the object's properties will be used to create an additional sql clause. Example:
  144.      *
  145.      * <code>
  146.      * $authors = author::load_all_by_publisher( 'Penguin', $include = array('book' => (object) array('year' => 1972)) );
  147.      * </code>
  148.      *
  149.      *
  150.      * @param array  $objects       Parent objects to attach the associated models to
  151.      * @param array  $associations  Class names of the associated models to be loaded
  152.      * @return array 
  153.      */
  154.     public static function load_associations($objects, $associations) {
  155.         global $CFG, $soda_module_name;
  156.         if (static::no_more_associations_to_load($associations)) return $objects;
  157.         $model_name = get_called_class();
  158.         foreach($associations as $parent_model => $child_model) {
  159.             if ( (is_string($parent_model)) && (strstr($parent_model, ':')) ) continue;
  160.             if ($directive = static::get_directive($child_model)) {
  161.                 $method = "call_{$directive}";
  162.                 static::$method( $objects, $parent_model, $child_model );
  163.                 continue;
  164.             }
  165.             if (is_string($parent_model)) {
  166.                 include_once("{$CFG->dirroot}/mod/{$soda_module_name}/models/{$parent_model}.php");
  167.                 // This should actually be done through the model class settings of the association class,
  168.                 // not as a parameter in the 'load' function call
  169.                 $foreign_key = static::construct_foreign_key($parent_model);
  170.                 $parents = $parent_model::load_all("{$foreign_key} IN (" . join(',', static::collect('id', $objects)) .  ")");
  171.                 if (!is_array($child_model)) $child_model = array($child_model);
  172.                 $parents = $parent_model::load_associations($parents, $child_model);
  173.                 static::attach_associations($objects, $parents, false, $foreign_key);
  174.                 continue;
  175.             }
  176.             include_once("{$CFG->dirroot}/mod/{$soda_module_name}/models/{$child_model}.php");
  177.             $foreign_key = static::construct_foreign_key($child_model);
  178.             $children = $child_model::load_all("{$foreign_key} IN (" . join(',', static::collect('id', $objects)) .  ")");
  179.             $objects = static::attach_associations($objects, $children, false, $foreign_key);
  180.         }
  181.         return $objects;
  182.     } // function load_associations

  200.      * Creates and executes an sql 'where clause' to restrict the set of associated items.
  201.      *
  202.      * @param array  $objects           Array of 'parent' objects

  204.      * @param array  $child_model       Array of 'directives', including the actual scope

  207.     public static function call_scope($objects, $association_model, $child_model) {
  208.         global $CFG, $soda_module_name;
  209.  
  210.         //$restrictions = static::get_first($child_model[':scope']);
  211.         $restrictions = $child_model[':scope'];
  212.         include_once("{$CFG->dirroot}/mod/{$soda_module_name}/models/{$association_model}.php");
  213.         $order = (isset($child_model[':order'])) ? " ORDER BY {$child_model[':order']} " : "";
  214.         $association_name = (isset($child_model[':association_name'])) ? $child_model[':association_name'] : false;

  216.         $children = $association_model::load_all("{$foreign_key} IN (" . join(',', static::collect('id', $objects)) .  ")

  223.     /**
  224.      * Looks up the static property table_relations to see if there is a relationship
  225.      * with $association_model which includes a foreign key specification.
  226.      *
  227.      * @param string $association_model Name of the associated model
  228.      * @return string                   Returns the foreign key or false
  229.      */
  230.     public static function find_foreign_key_for($association_model) {
  231.         if (! $relations = static::$table_relations) return false;
  232.         $keys = array_keys($relations);
  233.         if (! is_array($relations[$keys[0]]) ) $relations = array($relations);
  234.         foreach($relations as $relation) {
  235.             if (!array_key_exists(':foreign_key', $relation)) continue;
  236.             if (array_key_exists($association_model, array_flip($relation))) return $relation[':foreign_key'];
  237.         }               
  238.         return false;
  239.     } // function find_foreign_key_for

  243.      * Looks up the foreign key of the currenct model used in $association_model or

  249.     public static function construct_foreign_key($association_model) {

  251.         $model_name = get_called_class();
  252.         if ($foreign_key = static::find_foreign_key_for($association_model)) return $foreign_key;

  256.     } // function construct_foreign_key

  280.      * Extracts 'directive' out of $child_model array.

  285.     public static function get_directive($child_model) {
  286.         if (!is_array($child_model)) return false;        
  287.         foreach($child_model as $key => $value) {
  288.             if (!strstr($key, ':')) continue;
  289.             if (in_array(substr($key, 1), static::$directives) ) return substr($key, 1);
  290.         }
  291.         return false;
  292.     } // function get_directive

  295.     /**
  296.      * Loads associated models by if they appear in an association table together with
  297.      * an object out of $objects.
  298.      *
  299.      * @param array  $objects     Array of objects for which to find the associated models
  300.      * @param string $association Name of the association model
  301.      * @param string $through     Name of the combining model (i.e. the association table)
  302.      * @param string $order       Property by which to sort the associated object
  303.      * @return void 
  304.      */
  305.     public static function load_association_through($objects, $association, $through, $order = false) {
  306.         global $CFG, $soda_module_name;
  307.         $model_name = get_called_class();
  308.         include_once("{$CFG->dirroot}/mod/{$soda_module_name}/models/{$association}.php");
  309.         include_once("{$CFG->dirroot}/mod/{$soda_module_name}/models/{$through}.php");
  310.  
  311.         $through_objects = $through::load_all("{$model_name}_id IN (" . join(',', static::collect('id', $objects)) .  ")" );
  312.         $association_objects = $association::load_all("id IN (" . join(',', static::collect("{$association}_id", $through_objects)) .  ")" );
  313.         //exit(print_object($through_objects));
  314.  
  315.         $association_name = $association::plural();
  316.         $model_id_name = $model_name . '_id';
  317.         $association_id_name = $association . '_id';
  318.  
  319.         foreach($objects as $object) {
  320.             $selected = array();
  321.             foreach($through_objects as $through_object) {
  322.                 if ($object->id != $through_object->$model_id_name) continue;
  323.                 $selected[] = $association::find_by_id($through_object->$association_id_name, $association_objects);
  324.             }
  325.             $object->$association_name = $selected;
  326.             static::sort_by($object->$association_name, $order);
  327.         }
  328.     } // function load_association_through

  331.     /**
  332.      * Sorts an array of objects by a property or method.
  333.      *
  334.      * @param array  $objects            Array of objects to sort
  335.      * @param string $property_or_method Name of the property or method to sort by
  336.      * @param string $order              Sorting order (defaults to 'ascending', e.g. a, .. , z or 0, .. n)
  337.      * @return array                     Returns sorted array
  338.      */
  339.     public static function sort_by(&$objects, $property_or_method, $order = "ASC") {
  340.         if (! $property_or_method) return;
  341.         $keys = array_keys($objects);
  342.         if (! count($keys) ) return $objects;
  343.         $obj = $objects[$keys[0]];
  344.  
  345.         if (property_exists($obj, $property_or_method)) return static::sort_by_property($objects, $property_or_method, $order);
  346.         if (method_exists($obj, $property_or_method)) return static::sort_by_method($objects, $property_or_method, $order);
  347.     } // function sort_by

  351.      * Sorts an array of objects by a property
  352.      *
  353.      * @param array  $objects            Array of objects to sort
  354.      * @param string $property           Name of the property to sort by
  355.      * @param string $order              Sorting order (defaults to 'ascending', e.g. a, .. , z or 0, .. n)
  356.      * @return array                     Returns sorted array
  357.      */
  358.     public static function sort_by_property(&$objects, $property, $order) {
  359.         return usort($objects, function($a, $b) use ($property, $order) {
  360.             if ($a->$property == $b->$property ) return 0;
  361.             if ($order == 'ASC') {
  362.                 return ($a->$property < $b->$property) ? -1 : 1;
  363.             } else {
  364.                 return ($a->$property > $b->$property) ? -1 : 1;
  365.             }
  366.         });               
  367.     } // function sort_by_property

  391.     /**
  392.      * Attaches associated child objects to parent objects.
  393.      *
  394.      * Example: seat::load_associations($seats, $bookings)
  395.      *
  396.      * This will endow each 'seat' object with the property bookings.
  397.      * If a particular seat has no bookings, the property will be an empty array.
  398.      * Each booking, in turn, will be endowed with a property 'seat', pointing to the parent object.
  399.      *
  400.      * The associations parameter can also be a nested array:
  401.      * 
  402.      * <code>
  403.      * array('author' => array('book' => 'chapter'))
  404.      * </code>
  405.      *
  406.      * This 'include' array will result in an even deeper object hierarchy.
  407.      *
  408.      * If don't specify the optional association name, the pluralized name of the child model will be used as
  409.      * the property name.
  410.      * Alternatively, if you do specify a name, and the name matches the class name of the child model,
  411.      * (i.e. is singular), then exactly one associated object will be attached (many-to-one relation will be assumed).
  412.      *
  413.      * Example:
  414.      * <code>
  415.      * book::attach_associations($books, $authors, 'author');
  416.      * </code>
  417.      *
  418.      * This will create a property 'author' for each book, containing the associated author object
  419.      *
  420.      * @param array  $objects           Parent objects to attach the associated models to
  421.      * @param array  $children          Associated objects
  422.      * @param string $association_name  Name of the parent's property for the associated objects (optional)
  423.      * @return array 
  424.      */
  425.     public static function attach_associations($objects, $children, $association_name = false, $foreign_key = false) {
  426.         if ( (!is_array($objects)) || (!count($objects)) ) return false;
  427.         if ( (!is_array($children)) || (!count($children)) ) return false;
  428.         if (is_array(static::get_first($objects)) ) $objects = static::flatten($objects);
  429.         $model_name = get_called_class();
  430.         $foreign_key = ($foreign_key) ? $foreign_key : $model_name . '_id';
  431.         $finder = "find_all_by_{$foreign_key}";
  432.         $child_model = get_class(static::get_first($children));
  433.         if (!$association_name) $association_name = $child_model::plural();
  434.         if ($association_name == $child_model) $finder = "find_by_{$foreign_key}";
  435.         foreach($objects as $object) {
  436.             $object->$association_name = $child_model::$finder($object->id, $children);
  437.         }               
  438.         foreach($children as $child) {
  439.             $child->$model_name = $model_name::find_by_id($child->$foreign_key, $objects);
  440.         }
  441.         return $objects;
  442.     } // function attach_associations

  458.     /**
  459.      * Loads a collection of parent objects by association with a collection of child objects.
  460.      * For each child object, this function will attempt to look for the associated
  461.      * parent object. The child objects are inserted into an appropriately named property
  462.      * of the parent object.
  463.      * Each child object will also be endowed with a property named after the class of the parent object.
  464.      * This property points to the parent object.
  465.      *
  466.      * Example:
  467.      * <code>
  468.      * $authors = author::associative_load($books);
  469.      * </code>
  470.      *
  471.      * This will lookup all authors with the id $book->author_id and load them
  472.      * as a collection of author objects.
  473.      *
  474.      * Each author object will be endowed with the collection of associated book objects:
  475.      * $author->books
  476.      *
  477.      * And each book object will be endowed with an author property, pointing to the author object:
  478.      * $book->author
  479.      *
  480.      * @param array $associated_collection  collection which contains objects with parent_ids
  481.      * @param array $association_name       name for the property where the child objects will reside (optional)
  482.      * @return array 
  483.      */
  484.     public static function associative_load($associated_collection, $association_name = false) {
  485.         if ( (!is_array($associated_collection)) || (!count($associated_collection)) ) return false;
  486.         $model_name = get_called_class();
  487.         $child_model = get_class(static::get_first($associated_collection));
  488.         $objects = $model_name::load_all( "id IN (" . join(',', array_unique($child_model::collect($model_name . '_id', $associated_collection)) ) .")" );
  489.         return $model_name::attach_associations($objects, $associated_collection, $association_name);
  490.     } // function associative_load

  493.     /**
  494.      * Loads all objects of the table specified in model::$table_name
  495.      * For example, if your model looks like this:
  496.      * <code>
  497.      * class user extends model {
  498.      *   static $table_name = 'user';
  499.      * }
  500.      * </code>
  501.      *
  502.      * then calling $users = user::load_all() will return all users in the database as an array of Soda model objects.
  503.      *
  504.      *
  505.      * This method uses an optional array of class names to load associated objects.
  506.      * 
  507.      * Example: $authors = author::load_all(false, $include = array('book'));
  508.      * 
  509.      * You can also specify nested relations:
  510.      * 
  511.      * $publishers = publishers::load_all(false, $include = array('author' => 'book'));
  512.      * 
  513.      * This will load all authors and their books as an object hierarchy residing in each publisher object.
  514.      *
  515.      * Please note that there's also a 'singular' version of this function: model::load.
  516.      * 
  517.      *
  518.      * @param string $where_clause      SQL where clause: use without the WHERE keyword (optional)
  519.      * @param array  $include           Class names of associated models to include (optional)
  520.      * @param array  $params            Array of sql parameters (optional)
  521.      * @param int    $limitfrom         Return a subset of records, starting at this point (optional, required if $limitnum is set).
  522.      * @param int    $limitnum          Return a subset comprising this many records (optional, required if $limitfrom is set).
  523.      * @return array 
  524.      */
  525.     public static function load_all($where_clause = false, $include = false, $params = null, $limitfrom = null, $limitnum = null) {
  526.         global $CFG, $DB;
  527.         $where = ($where_clause) ? "WHERE $where_clause " : "";
  528.         if (! $recordset = $DB->get_recordset_sql("SELECT *
  529.                                                    FROM {$CFG->prefix}" . static::$table_name . " 
  530.                                                    $where",
  531.                                                    $params,
  532.                                                    $limitfrom, $limitnum) ) return;
  533.         $objects = static::convert_to_objects($recordset);
  534.         $recordset->close();

  536.         return $objects;       
  537.     } // function load_all

  697.     /**
  698.      * Builds an associative array by using an array of property names as the keys and an array of arguments
  699.      * as the values.
  700.      * This method is typically used to deconstruct a dynamic finder or loader call into a SQL where
  701.      * clause which is used in default load or find call.
  702.      *
  703.      * @param  array       $property_names   Array of column names
  704.      * @param  array       $args             Array of values for the columns
  705.      * @return array                         Returns an associative array of column names pointing to argument values
  706.      */
  707.     public static function map_properties_to_values($property_names, $args) {
  708.         $properties = array();
  709.         for($i = 0; $i < count($property_names); $i++) {
  710.             $properties[$property_names[$i]] = $args[$i];
