  41.     /**
  42.      * Reroutes method calls to non-existing methods to controller methods.
  43.      * Throws an exception if the called method does not exist as a controller method either.
  44.      * Please note that this behavior is symmetrical: the Soda controller class reroutes calls to unknown
  45.      * methods to the helper class.
  46.      *
  47.      * @param  string       $method Name of the method to call dynamically
  48.      * @param  array        $args   Array of arguments with the method
  49.      * @return mixed        Returns whatever the called method returns
  50.      */
  51.     function __call($method, $args) {    
  52.  
  53.         if (method_exists($this->controller, $method)) {
  54.             return call_user_func_array(
  55.                 array($this->controller, $method),
  56.                 $args
  57.             );
  58.         }
  59.         foreach($this->controller->helpers as $helper) {
  60.             if (!method_exists($helper, $method)) continue;
  61.             return call_user_func_array(
  62.                 array($helper, $method),
  63.                 $args
  64.             );
  65.  
  66.         }
  67.         throw new Exception("Unknown method [$method]");
  68.     } // function __call

  71.     /**
  72.      * Reroutes method calls to non-existing properties to controller properties.
  73.      * Throws an exception if the called property does not exist as a controller property either.
  74.      * Please note that this behavior does not display the same symmetry as __call. There is no corresponding
  75.      * method in the controller class.
  76.      *
  77.      * @param  string       $method Name of the property to call dynamically
  78.      * @param  array        $args   Array of arguments with the property
  79.      * @return mixed        Returns whatever the called property returns
  80.      */
  81.     function __get($property) {
  82.         if (property_exists($this->controller, $property)) {
  83.             return $this->controller->$property;
  84.         }
  85.         $class = get_class($this->controller);
  86.         if (property_exists($class, $property)) {
  87.             return $class::$property;
  88.         }
  89.         throw new Exception("Unknown property [$property]");
  90.     } // function __get

  93.     /**
  94.      * Prints opening form tag and hidden form fields to specify the destination of a form post.
  95.      * If you do not specify the action as part of the post parameters, the form post will be
  96.      * handled by the default 'save' method of the current controller.
  97.      * You can also specify additional post parameters to change the default behavior of the form post. Example:
  98.      *
  99.      * <code>
  100.      * $this->form_open(false, $post_parameters = array('controller' => 'some_alternative_controller'));
  101.      * </code>
  102.      *
  103.      * The third argument lets you overwrite the default html attributes of the form tag. The defaults are:
  104.      * - action: $this->base_url
  105.      * - method: 'post'
  106.      * - name:   $this->model_name
  107.      * - class:  'mform'
  108.      *
  109.      * @param  string       $action          Controller method to handle the form post, defaults to 'save' (optional)
  110.      * @param  array        $post_parameters Array of variables to submit through hidden form fields (optional)
  111.      * @param  array        $html_attributes Array of attributes to include in form tag (optional)
  112.      * @return void 
  113.      */
  114.     function form_open($action = 'save', $post_parameters = array(), $html_attributes = array() ) {
  115.         $this->form_tag($html_attributes);
  116.         if (!array_key_exists('action', $post_parameters)) $post_parameters['action'] = $action;
  117.         echo $this->create_hidden_fields($post_parameters);
  118.     } // function form

  121.     /**
  122.      * Prints opening form tag.
  123.      * 
  124.      * The optional argument lets you overwrite the default html attributes of the form tag. The defaults are:
  125.      * - action: $this->base_url
  126.      * - method: 'post'
  127.      * - name:   $this->model_name
  128.      * - class:  'mform'
  129.      *
  130.      * @param  array        $html_attributes Array of attributes to include in form tag (optional)
  131.      * @return void 
  132.      */
  133.     function form_tag( $html_attributes = array() ) {
  134.         if (!array_key_exists('action', $html_attributes)) $html_attributes['action'] = $this->base_url;
  135.         if (!array_key_exists('method', $html_attributes)) $html_attributes['method'] = 'post';
  136.         if (!array_key_exists('name', $html_attributes)) $html_attributes['name'] = $this->model_name;
  137.         if (!array_key_exists('class', $html_attributes)) $html_attributes['class'] = 'mform';
  138.         echo "<form {$this->create_html_attributes($html_attributes)}>";
  139.     } // function form_tag

  142.     function create_html_attributes($attributes) {
  143.         if (! is_array($attributes) || !count($attributes) ) return "";
  144.         $pairs = array();
  145.         foreach($attributes as $name => $value) {
  146.             $pairs[] = "{$name}='{$value}'";
  147.         }
  148.         return join(' ', $pairs);
  149.     } // function create_html_attributes

  152.     /**
  153.      * Returns path to a partial view.
  154.      * A partial is a part of a view, typically reused in multiple places within the same view, or shared by multiple views.
  155.      * The method will first lookup the most specific path, i.e. by looking in the views directory for the current model.
  156.      * If the partial cannot be found there, the method will lookup the views/shared path.
  157.      *
  158.      * @param   string  $partial    Name of the partial
  159.      * @return                      Returns string
  160.      */
  161.     function get_partial_path($partial) {
  162.         global $CFG, $soda_module_name;
  163.  
  164.         if (file_exists("{$CFG->dirroot}/mod/$soda_module_name/views/{$this->model_name}/_{$partial}.html")) {
  165.             return "{$CFG->dirroot}/mod/$soda_module_name/views/{$this->model_name}/_{$partial}.html";
  166.         }
  167.         return "{$CFG->dirroot}/mod/$soda_module_name/views/shared/_{$partial}.html";
  168.     } // function get_partial_path
