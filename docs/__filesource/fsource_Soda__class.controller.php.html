  152.     /**
  153.      * Reroutes method calls to non-existing methods to helper methods.
  154.      * Throws an exception if the called method does not exist as a helper method either.
  155.      * Please note that this behavior is symmetrical: the Soda helper class reroutes calls to unknown
  156.      * methods to the controller class.
  157.      *
  158.      * @param  string       $method Name of the method to call dynamically
  159.      * @param  array        $args   Array of arguments with the method
  160.      * @return mixed        Returns whatever the called method returns
  161.      */
  162.     function __call($method, $args) {
  163.         foreach($this->helpers as $helper) {
  164.             if (!method_exists($helper, $method)) continue;
  165.             return call_user_func_array(
  166.                 array($helper, $method),
  167.                 $args
  168.             );
  169.         }
  170.         throw new Exception("Unknown method [$method]");
  171.     } // function __call

  240. /**

  255.      *                                         The messages will be displayed in the view associated with $action

  371.     /**
  372.      * Creates a series of hidden input fields for inclusion in a form.
  373.      * If you omit the parameters array, the Soda action will be set to 'save'. Additionally, a controller and id
  374.      * hidden input field will be created. The default value for controller is the name of the current controller.
  375.      * The id parameter is the Moodle module's instance id.
  376.      *
  377.      *
  378.      * @param  array  $parameters  Associative array to be transformed in hidden input fields (optional)

  380.      */
  381.     function create_hidden_fields($parameters = array() ) {
  382.         global $id;
  383.         if (! array_key_exists('controller', $parameters) ) $parameters['controller'] = $this->model_name;
  384.         if (! array_key_exists('id', $parameters) ) $parameters['id'] = $id;
  385.         if (! array_key_exists('action', $parameters)) $parameters['action'] = 'save';
  386.         $inputs = array();
  387.         foreach($parameters as $key => $value) {
  388.             $inputs[] = "<input type='hidden' value='$value' name='$key' id='$key'/>";
  389.         }
  390.         return join("\n", $inputs);
  391.     } // function create_hidden_fields
