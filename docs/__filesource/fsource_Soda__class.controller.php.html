  158.     /**
  159.      * Reroutes method calls to non-existing methods to helper methods.
  160.      * Throws an exception if the called method does not exist as a helper method either.
  161.      * Please note that this behavior is symmetrical: the Soda helper class reroutes calls to unknown
  162.      * methods to the controller class.
  163.      *
  164.      * @param  string       $method Name of the method to call dynamically
  165.      * @param  array        $args   Array of arguments with the method
  166.      * @return mixed        Returns whatever the called method returns
  167.      */
  168.     function __call($method, $args) {
  169.         foreach($this->helpers as $helper) {
  170.             if (!method_exists($helper, $method)) continue;
  171.             return call_user_func_array(
  172.                 array($helper, $method),
  173.                 $args
  174.             );
  175.         }
  176.         throw new Exception("Unknown method [$method]");
  177.     } // function __call

  304.     /**

  307.      * You can also provide an associative array as the first argument. The keys in this array will made available

  316.     function get_view($data_array = array(), $view = false) {
  317.         global $CFG, $id;
  318.  
  319.         foreach($data_array as $variable_name => $value) {
  320.             $$variable_name = $value;
  321.         }
  322.         //include correct view
  323.         $trace = debug_backtrace();
  324.         $this->view = ($view) ? $view : $trace[1]['function'];
  325.         $view_path = "{$CFG->dirroot}/mod/{$this->mod_name}/views/{$this->model_name}/{$this->view}.html";
  326.         if (! file_exists($view_path)) {
  327.             $parent_views = static::model_name(get_parent_class($this));
  328.             $view_path = "{$CFG->dirroot}/mod/{$this->mod_name}/views/{$parent_views}/{$this->view}.html";
  329.         }
  330.         $template_path = "{$CFG->dirroot}/mod/$this->mod_name/views/template.html";
  331.         if (file_exists($template_path)) {
  332.             return include_once($template_path);

  335.     } // function get_view
