  32. /**
  33.  * Class soda_error stores error messages meant to be displayed for end users.
  34.  * A typical use case is validation. If a model is not valid, the error messages
  35.  * is added to soda_error::$validation errors. You can display the invalid form and
  36.  * use the error messages to show the user which fields are "in error".
  37.  * @package Soda
  38.  */
  39. class soda_error {
  40.  
  41.     static $validation_errors = array();
  42.  
  43.  
  44.     /**
  45.      * Returns true if soda_error::$validation_errors has at least one entry.
  46.      *
  47.      * @return  boolean Returns true if soda_error::$validation_errors has at least one entry, otherwise false.
  48.      */
  49.     static function invalid() {
  50.         return (count(self::$validation_errors));
  51.     } // function invalid
  52.  
  53.  
  54.     /**
  55.      * Returns true if soda_error::$validation_errors has no entries
  56.      *
  57.      * @return  boolean Returns true if soda_error::$validation_errors has no entries, otherwise false.
  58.      */
  59.     static function valid() {
  60.         return !self::invalid();        
  61.     } // function valid
  62.  
  63.  
  64.     /**
  65.      * Adds an error message to soda_error::$validation_errors for the given object and field.
  66.      *
  67.      * @param   object  $object     Soda model that contains an error
  68.      * @param   string  $field      Field that is in error
  69.      * @param   string  $message    Error message for user
  70.      * @return  void 
  71.      */
  72.     static function add_error($object, $field, $message) {
  73.         $model_name = get_class($object);
  74.         if ( (isset(self::$validation_errors[$model_name][spl_object_hash($object)][$field])) && ( in_array($message, self::$validation_errors[$model_name][spl_object_hash($object)][$field]) ) ) return;
  75.         self::$validation_errors[$model_name][spl_object_hash($object)][$field][] = $message;
  76.         //return print_object(self::$validation_errors);
  77.     } // function add_error
  78.  
  79.  
  80.     /**
  81.      * Determines whether a given object is invalid for a given field.
  82.      * If you omit the field, this function will return true if there is at least one error for the object.
  83.      *
  84.      * @param   object  $object     Soda model
  85.      * @param   string  $field      Field
  86.      * @return  boolean             Returns true if the object is in error, otherwise false
  87.      */
  88.     static function in_error($object, $field = false) {
  89.         $model_name = get_class($object);
  90.         if (!isset(self::$validation_errors[$model_name])) return false;
  91.         if (!isset(self::$validation_errors[$model_name][spl_object_hash($object)])) return false;
  92.         //print_object(self::$validation_errors[$model_name][spl_object_hash($object)]);
  93.         if (!$field) return true;
  94.         if (!isset(self::$validation_errors[$model_name][spl_object_hash($object)][$field])) return false;
  95.         return true;
  96.     } // function in_error
  97.  
  98.  
  99.     static function class_in_error($model_name) {
  100.         return isset(self::$validation_errors[$model_name]);
  101.     } // function class_in_error
  102.  
  103.     
  104.     /**
  105.      * Returns the first error of the given object and field.
  106.      * If there are no errors, the function returns false.
  107.      *
  108.      * @param   object  $object     Soda model
  109.      * @param   string  $field      Name of the field
  110.      * @return  boolean             Returns the first error if there is one, otherwise false
  111.      */
  112.     static function get_first_error($object, $field) {
  113.         $model_name = get_class($object);
  114.         if (!isset(self::$validation_errors[$model_name][spl_object_hash($object)][$field])) return false;
  115.         $keys = array_keys(self::$validation_errors[$model_name][spl_object_hash($object)][$field]);
  116.         return (self::in_error($object, $field)) ? self::$validation_errors[$model_name][spl_object_hash($object)][$field][$keys[0]] : false;
  117.     } // function get_first_error
  118.  
  119.  
  120.     /**
  121.      * Calls an anonymous function for each error.
  122.      * The anonymous function should have the following call signature:
  123.      *
  124.      *   $model_name, $instance, $field,  $message
  125.      *
  126.      * @param   function  $function The function to call for each error.     Soda model
  127.      * @return  void 
  128.      */
  129.     static function foreach_error($function) {
  130.         foreach(self::$validation_errors as $model_name => $instances) {
  131.             foreach($instances as $instance => $fields) {
  132.                 foreach($fields as $field => $messages) {
  133.                     foreach($messages as $message) {
  134.                         $function($model_name, $instance, $field, $message);
  135.                     }
  136.                 }
  137.             }
  138.         }
  139.     } // function foreach_error
  140.  
  141. } // class soda_error
