  37. /**
  38.  * The Soda base class. If you instantiate this class, all default lib functions for
  39.  * Moodle module will be made available automatically. These are accessible as both
  40.  * static functions on your Soda derived module class and as global functions prefixed
  41.  * with your module's name.
  42.  *
  43.  * To build a Moodle module with Soda, you need at least three files. Here, we'll provide
  44.  * an example derived from a module called 'compass':
  45.  *
  46.  *   - mod/compass/index.php
  47.  *   - mod/compass/lib.php
  48.  *   - mod/compass/class.compass.php
  49.  *
  50.  * The first file, index.php should look like this:
  51.  * <code>
  52.  *   # mod/compass/index.php
  53.  *   require_once("../../config.php");
  54.  *   include_once('lib.php');
  55.  *   $compass_instance->display();
  56.  * </code>
  57.  * 
  58.  * Apart from these lines of code, your index.php file can be entirely empty.
  59.  *
  60.  * The $compass_instance variable is assigned in lib.php:
  61.  * <code>
  62.  *   # mod/compass/lib.php
  63.  *   require_once("class.compass.php");
  64.  *   $compass_instance = new compass();
  65.  * </code>
  66.  *
  67.  * Again, apart from these lines of code, your lib.php file can be entirely empty.
  68.  *
  69.  *
  70.  * class.compass.php, in turn, looks like this:
  71.  * <code>
  72.  *   # mod/compass/class.compass.php
  73.  *   include_once("{$CFG->dirroot}/local/soda/class.soda.php");
  74.  *   class compass extends soda {}
  75.  * </code>
  76.  *
  77.  * Please use the soda extending class (i.e. class compass in our example) to overwrite any non-default functions
  78.  * you would normally place in lib.php.
  79.  *
  80.  *
  81.  * Once these classes are in place, you can call display() on your module's Soda extension in your module's index.php file:
  82.  *
  83.  *   $compass_instance->display();

  120.     /**
  121.      * Dispatches control to the Soda based Module.
  122.      * The dispatcher instantiates the appropriate controller, model and helper classes.
  123.      * If there is no 'controller' parameter, Soda will use module name to find a controller.
  124.      * In case there is no [module_name]_controller either, the Soda controller (this class) will
  125.      * be instantiated.
  126.      * Then, the method from the $action parameter is invoked (the action function is called).
  127.      * Finally, the after_action method is called on the controller.
  128.      *
  129.      * @param   string  $action Method to call on the target controller
  130.      * @return  void 
  131.      */
  132.     function dispatch($action) {
  133.         $mod_name = get_called_class();
  134.         global $CFG, ${$mod_name}, $PAGE, $soda_module_name, $cm;
  135.  
  136.         $controller = optional_param('controller', $mod_name, PARAM_RAW);
  137.  
  138.  
  139.  
  140.         $general_helper = $this->get_helper($mod_name);
  141.         $specific_helper  = $this->get_helper($mod_name, $controller);
  142.  
  143.         if (! controller::load_file($controller, $mod_name)) {
  144.             $instance = new controller($mod_name, ${$mod_name}->id, $action);
  145.             $instance->set_helpers(array($general_helper, $specific_helper));
  146.             return $instance->$action();
  147.         }
  148.         $record_id = optional_param("{$controller}_id", false, PARAM_INT);
  149.         if (file_exists("{$CFG->dirroot}/mod/$mod_name/models/{$controller}.php")) {
  150.             include_once("{$CFG->dirroot}/mod/$mod_name/models/{$controller}.php");
  151.         }
  152.         $class = $controller . "_controller";
  153.         $instance = new $class($mod_name, ${$mod_name}->id, $action);
  154.         $instance->set_helpers(array($general_helper, $specific_helper));
  155.         $instance->$action($record_id);               
  156.         $this->redirect = $instance->redirect;
  157.         $instance->after_action();
  158.     } // function dispatch

  191.         $mod_name = get_called_class();
  192.         global $CFG, $cm, $PAGE, $soda_module_name, ${$mod_name}, $course;

  194.         // Included to fix problem: "Coding problem: $PAGE->context was not set. You may have forgotten to call

  203.  
  204.         if ($this->no_layout_required() ) {
  205.             echo $content;
  206.             return;
  207.         }

  209.         $PAGE->set_pagelayout('admin');
  210.  
  211.         $header = $this->get_header(get_called_class());
  212.  
  213.         echo $header;

  215.         $this->print_footer(get_called_class());

  268.     /**
  269.      * Creates globally defined wrapper functions for all static functions in this class.
  270.      * Uses the 'mod_name' to prefix the functions to create unique function names.
  271.      * This eliminates the need to specify the same functions for each new Moodle module in lib.php.
  272.      *                                                                                   
  273.      * Example: if your mod is called 'planner', planner_get_instance will be created
  274.      * as a wrapper for planner::get_instance() - which will typically be inherited. I.e. there is a soda::get_instance().
  275.      *
  276.      * @param   string  $mod_name   Name of the module
  277.      * @return  void 
  278.      */
  279.     function create_mod_library_functions($mod_name) {
  280.         $reflection = new ReflectionClass( get_called_class() );
  281.         foreach($reflection->getMethods(ReflectionMethod::IS_STATIC) as $method) {
  282.             $this->create_wrapper_function($method, $mod_name);
  283.         }
  284.         // special case deviating from naming convention
  285.         if (! function_exists($mod_name . '_get_' . $mod_name) ) {
  286.             eval('function ' . $mod_name . '_get_' . $mod_name . '($mod_id) { return ' . $mod_name . '::get_mod_by_id($mod_id); }');
  287.         }
  288.     } // function create_mod_library_functions

  291.     /**
  292.      * Creates a globally defined wrapper function.
  293.      *
  294.      * @param   string  $method     Reference to the method
  295.      * @param   string  $mod_name   Name of the module
  296.      * @return  void 
  297.      */
  298.     function create_wrapper_function($method, $mod_name) {
  299.         $base_method_name = $method->name;
  300.         $param_names = array();
  301.         $params = $method->getParameters();
  302.         foreach ($params as $param) {
  303.             $param_names[] = '$' . $param->getName();
  304.         }
  305.         $arguments = (count($param_names)) ? join(', ', $param_names) : '';
  306.         $function_name = $mod_name . '_' . $base_method_name;
  307.         if (! function_exists($function_name)) {
  308.             $str = "function $function_name($arguments) { return $mod_name::$base_method_name($arguments); }";
  309.             // Better close your eyes now...
  310.             eval($str);
  311.         }               
  312.     } // function create_wrapper_function
