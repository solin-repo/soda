  37. /**
  38.  * The Soda base class. If you instantiate this class, all default lib functions for
  39.  * Moodle module will be made available automatically. These are accessible as both
  40.  * static functions on your Soda derived module class and as global functions prefixed
  41.  * with your module's name.
  42.  *
  43.  * To build a Moodle module with Soda, you need at least three files. Here, we'll provide
  44.  * an example derived from a module called 'compass':
  45.  *
  46.  *   - mod/compass/index.php
  47.  *   - mod/compass/lib.php
  48.  *   - mod/compass/class.compass.php
  49.  *
  50.  * The first file, index.php should look like this:
  51.  * <code>
  52.  *   # mod/compass/index.php
  53.  *   require_once("../../config.php");
  54.  *   include_once('lib.php');
  55.  *   $compass_instance->display();
  56.  * </code>
  57.  * 
  58.  * Apart from these lines of code, your index.php file can be entirely empty.
  59.  *
  60.  * The $compass_instance variable is assigned in lib.php:
  61.  * <code>
  62.  *   # mod/compass/lib.php
  63.  *   require_once("class.compass.php");
  64.  *   $compass_instance = new compass();
  65.  * </code>
  66.  *
  67.  * Again, apart from these lines of code, your lib.php file can be entirely empty.
  68.  *
  69.  *
  70.  * class.compass.php, in turn, looks like this:
  71.  * <code>
  72.  *   # mod/compass/class.compass.php
  73.  *   include_once("{$CFG->dirroot}/mod/soda/class.soda.php");
  74.  *   class compass extends soda {}
  75.  * </code>
  76.  *
  77.  * Please use the soda extending class (i.e. class compass in our example) to overwrite any non-default functions
  78.  * you would normally place in lib.php.
  79.  *
  80.  *
  81.  * Once these classes are in place, you can call display() on your module's Soda extension in your module's index.php file:
  82.  *
  83.  *   $compass_instance->display();

  120.     /**
  121.      * Dispatches control to the Soda based Module.
  122.      * The dispatcher instantiates the appropriate controller, model and helper classes.
  123.      * If there is no 'controller' parameter, Soda will use module name to find a controller.
  124.      * In case there is no [module_name]_controller either, the Soda controller (this class) will
  125.      * be instantiated.
  126.      * Then, the method from the $action parameter is invoked (the action function is called).
  127.      * Finally, the after_action method is called on the controller.
  128.      *
  129.      * @param   string  $action Method to call on the target controller
  130.      * @return  void 
  131.      */
  132.     function dispatch($action) {
  133.         $mod_name = get_called_class();
  134.         global $CFG, ${$mod_name};
  135.  
  136.         $controller = optional_param('controller', $mod_name, PARAM_RAW);
  137.  
  138.         $general_helper = $this->get_helper($mod_name);
  139.         $specific_helper  = $this->get_helper($mod_name, $controller);
  140.  
  141.         if (! controller::load_file($controller, $mod_name)) {
  142.             $instance = new controller($mod_name, ${$mod_name}->id, $action);
  143.             $instance->set_helpers(array($general_helper, $specific_helper));
  144.             return $instance->$action();
  145.         }
  146.         $record_id = optional_param("{$controller}_id", false, PARAM_INT);
  147.         if (file_exists("{$CFG->dirroot}/mod/$mod_name/models/{$controller}.php")) {
  148.             include_once("{$CFG->dirroot}/mod/$mod_name/models/{$controller}.php");
  149.         }
  150.         $class = $controller . "_controller";
  151.         $instance = new $class($mod_name, ${$mod_name}->id, $action);
  152.         $instance->set_helpers(array($general_helper, $specific_helper));
  153.         $instance->$action($record_id);               
  154.         $this->redirect = $instance->redirect;
  155.         $instance->after_action();
  156.     } // function dispatch

  181.     /**
  182.      * Outputs the result of the action called on the controller and conditionally renders the layout as well.
  183.      * By default, whenever you call a Soda controller's redirect method, no layout will be rendered.
  184.      *

  187.      */
  188.     function add_layout_and_dispatch($action) {
  189.         global $CFG;
  190.  
  191.         ob_start(); // Start output buffering
  192.         $this->dispatch($action);
  193.         $content = ob_get_contents(); // Store buffer in variable
  194.         ob_end_clean(); // End buffering and clean up
  195.  
  196.         if ($this->no_layout_required() ) {
  197.             echo $content;
  198.             return;
  199.         }
  200.         $header = $this->get_header(get_called_class());
  201.  
  202.         echo $header;
  203.         echo $content;
  204.         $this->print_footer(get_called_class());
  205.     } // function add_layout_and_dispatch

  253.     /**
  254.      * Creates globally defined wrapper functions for all static functions in this class.
  255.      * Uses the 'mod_name' to prefix the functions to create unique function names.
  256.      * This eliminates the need to specify the same functions for each new Moodle module in lib.php.
  257.      *                                                                                   
  258.      * Example: if your mod is called 'planner', planner_get_instance will be created
  259.      * as a wrapper for planner::get_instance() - which will typically be inherited. I.e. there is a soda::get_instance().
  260.      *
  261.      * @param   string  $mod_name   Name of the module
  262.      * @return  void 
  263.      */
  264.     function create_mod_library_functions($mod_name) {
  265.         $reflection = new ReflectionClass( get_called_class() );
  266.         foreach($reflection->getMethods(ReflectionMethod::IS_STATIC) as $method) {
  267.             $this->create_wrapper_function($method, $mod_name);
  268.         }
  269.         // special case deviating from naming convention
  270.         if (! function_exists($mod_name . '_get_' . $mod_name) ) {
  271.             eval('function ' . $mod_name . '_get_' . $mod_name . '($mod_id) { return ' . $mod_name . '::get_mod_by_id($mod_id); }');
  272.         }
  273.     } // function create_mod_library_functions

  276.     /**
  277.      * Creates a globally defined wrapper function.
  278.      *
  279.      * @param   string  $method     Reference to the method
  280.      * @param   string  $mod_name   Name of the module
  281.      * @return  void 
  282.      */
  283.     function create_wrapper_function($method, $mod_name) {
  284.         $base_method_name = $method->name;
  285.         $param_names = array();
  286.         $params = $method->getParameters();
  287.         foreach ($params as $param) {
  288.             $param_names[] = '$' . $param->getName();
  289.         }
  290.         $arguments = (count($param_names)) ? join(', ', $param_names) : '';
  291.         $function_name = $mod_name . '_' . $base_method_name;
  292.         if (! function_exists($function_name)) {
  293.             $str = "function $function_name($arguments) { return $mod_name::$base_method_name($arguments); }";
  294.             // Better close your eyes now...
  295.             eval($str);
  296.         }               
  297.     } // function create_wrapper_function
